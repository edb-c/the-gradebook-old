# Installation Steps need to run this locally

    $ git clone https://github.com/edb-c/the-gradebook
    $ cd the-gradebook
    $ bundle install
    $ rake db:create  
    $ rake db:migrate
    $ rake db:seed

# Run the server

    $ rails s
